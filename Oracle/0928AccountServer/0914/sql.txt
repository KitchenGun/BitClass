-- account 테이블 생성

-- account 테이블 삭제
drop table account;

-- 시퀀스 생성
CREATE SEQUENCE accid_seq 
INCREMENT BY 10 
START WITH 1000
NOCYCLE  
NOCACHE; 

-- account 테이블 insert( 시퀀스, 이름, 입금액, 현재날짜)
insert into account values(accid_seq.nextval,'홍길동' ,5000, sysdate);

-- 이름으로 계좌id 검색
select accid from  account where name = '홍길동';

-- 계좌id로 계좌 전체 정보 검색
select * from account where accid = 1510;
